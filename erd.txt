user
----
id integer PK
username varchar(32)
showedName varchar(32)
icon url FK >0- image.user_icon
rating integer
bio varchar(250)
email varchar(60)
phone varchar(8)
password varchar(30)
is_age18 boolean
is_admin boolean
created_at datetime
updated_at datetime


event
-----
id integer PK
host_id integer FK >0- user.id
eventPicture url FK >0- image.event_image
title varchar(60)
category varchar(20)
hashtag varchar(20)
date date
time datetime
cost integer
location varchar(20)
FAQ varchar(500)
is_age18 boolean
is_private boolean


event_participant
-----------------
id integer PK
event_id integer FK >0- event.id
user_id integer FK >0- user.id


event_rating
------------
id integer PK
event_id integer FK >0- event.id
user_id integer FK >0- user.id
rating rating
command varchar(250)
created_at datetime

image
-----
id integer PK
event_image url
user_icon url

event_timeslot
--------------
id integer
period date FK >0- event.date
timeslot datetime FK >0- event.time
